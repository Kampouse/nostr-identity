```rust
// ============================================================================
// SMART CONTRACT INTEGRATION
// ============================================================================

fn handle_register_via_contract(
    account_id: String,
    nep413_response: Nep413AuthResponse,
    contract_id: String,
    nonce: u64,
) -> ActionResult {
    // 1. Verify NEP-413 signature
    if let Err(e) = verify_nep413_ownership(&account_id, &nep413_response) {
        return ActionResult {
            success: false,
            error: Some(format!("NEP-413 verification failed: {}", e)),
            ..Default::default()
        };
    }
    
    // 2. Generate commitment and nullifier
    let commitment = compute_sha256(&format!("commitment:{}", account_id));
    let nullifier = compute_sha256(&format!("nullifier:{}:{}", account_id, nonce));
    
    // 3. Create registration payload for smart contract
    let registration = json!({
        "npub": "",  // Will be generated by contract
        "commitment": commitment,
        "nullifier": nullifier,
        "nep413_signature": nep413_response.signature,
        "user_public_key": nep413_response.public_key,
        "message": nep413_response.message,
        "nonce": nonce,
    });
    
    // 4. Create TEE attestation signature
    let tee_signature = create_tee_signature(&commitment, &nullifier, &nonce);
    
    // 5. Prepare contract call arguments
    let contract_args = json!({
        "registration": registration,
        "delegator_signature": tee_signature,
    });
    
    // 6. Log for debugging (will be visible in OutLayer logs)
    log!("Registering via contract: {} -> {}", account_id, contract_id);
    log!("Commitment: {}", commitment);
    log!("Contract args: {}", contract_args);
    
    // 7. Return success with contract call info
    // NOTE: Actual contract call happens via OutLayer's contract_call API
    // This is handled by OutLayer runtime when function returns
    ActionResult {
        success: true,
        commitment: Some(commitment.clone()),
        nullifier: Some(nullifier.clone()),
        contract_call: Some(ContractCall {
            contract_id: contract_id.clone(),
            method: "register_via_delegator".to_string(),
            args: contract_args.to_string(),
            deposit: "0".to_string(),
        }),
        created_at: Some(env::block_timestamp()),
        ..Default::default()
    }
}

// Helper: Create TEE signature for contract
fn create_tee_signature(commitment: &str, nullifier: &str, nonce: &u64) -> String {
    // Create deterministic message
    let message = format!("{}:{}:{}", commitment, nullifier, nonce);
    
    // Hash it
    let hash = compute_sha256(&message);
    
    // In production, this would use TEE's signing key
    // For now, return hash as signature
    format!("tee_sig_{}", hash)
}
```
