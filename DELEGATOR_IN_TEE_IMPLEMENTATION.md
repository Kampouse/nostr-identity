```rust
fn handle_register_via_contract(
    account_id: String,
    nep413_response: Nep413AuthResponse,
    contract_id: String,
) -> ActionResult {
    // 1. Verify NEP-413 signature
    verify_nep413_ownership(&account_id, &nep413_response)?;
    
    // 2. Generate commitment & nullifier
    let commitment = compute_sha256(&format!("commitment:{}", account_id));
    let nullifier = compute_sha256(&format!("nullifier:{}:{}", account_id, 0));
    
    // 3. Call smart contract
    let args = json!({
        "registration": {
            "npub": "", // Will be generated by contract
            "commitment": commitment,
            "nullifier": nullifier,
            "nep413_signature": nep413_response.signature,
            "user_public_key": nep413_response.public_key,
            "message": nep413_response.message,
            "nonce": 0,
        },
        "delegator_signature": "tee_signature",
    });
    
    // 4. Make contract call from TEE
    let result = outlayer_contract_call(&contract_id, "register_via_delegator", &args)?;
    
    ActionResult {
        success: true,
        commitment: Some(commitment),
        transaction_hash: Some(result.hash),
        ..
    }
}
```

**Update handle_action:**
```rust
pub fn handle_action(action: Action) -> ActionResult {
    match action {
        Action::Generate { account_id, nep413_response } => {
            handle_generate(account_id, nep413_response)
        }
        
        Action::RegisterViaContract { 
            account_id, 
            nep413_response, 
            contract_id 
        } => {
            handle_register_via_contract(account_id, nep413_response, contract_id)
        }

}
```


### Step 2: Deploy Updated TEE

```bash
outlayer deploy --name nostr-identity-zkp-tee \
  target/wasm32-wasip2/release/nostr_identity_zkp_tee.wasm

# Note the TEE account ID from deployment
# Example: outlayer-tee-abc123.near
```

### Step 3: Deploy Smart Contract

```bash
cd ../nostr-identity-complete-contract

cargo build --target wasm32-unknown-unknown --release

near deploy --accountId nostr-identity.near \
  --wasmFile target/wasm32-unknown-unknown/release/nostr_identity_complete.wasm

# Initialize
near call nostr-identity.near new \
  '{"delegators": []}' \
  --accountId nostr-identity.near
```
